And this is what would the api tool's custom configuration logic would look like:
____
// Import necessary functions or 
// Tools such as Axios...


const api = {
    name: 'Chatbot', // name of the api tool
    description: 'something something...', // what this api tool do
    route: '/ai', // route that will be used to access the api tool.
    
    // possible example of the parameter content
    params: {
        "q=": String, // q=hi how are you
        "uid=": int, // uid=1000
        "think=": boolean // true or false
    },
    
    category: 'AI', // automatically creates a new or existing folder in the project
    api-key: true // true if the api requires apikey to work. false for free access
}

// Assumed output of the api:
// localhost:9000/AI/ai?q=hi&uid=1000&think=true&apikey=bilyabits-1234567890

importAsset(api, {
    // code logic of this api tool
})
